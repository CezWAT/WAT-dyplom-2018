#include <stdio.h>

float napiecie;
int raw_i;
char raw[3];

int main(){
  while(1){
    fgets(raw, 5, fopen("/sys/bus/iio/devices/iio:device0/in_voltage0_raw", "r"));
    raw_i = atoi(raw);
    napiecie = (float) 2*raw_i/527,724;
    printf("Napiêcie: %.4f ADC (int): %d\n", napiecie, raw_i);
    sleep(1);
  }
  return 0;
}
